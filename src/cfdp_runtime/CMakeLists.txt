file(
    GLOB
    HEADER_LIST
    CONFIGURE_DEPENDS
    "${cfdp_SOURCE_DIR}/include/runtime/cfdp_runtime/*.hpp"
)

    cfdp_runtime
    PUBLIC
    "${cfdp_SOURCE_DIR}/include/runtime"
)
